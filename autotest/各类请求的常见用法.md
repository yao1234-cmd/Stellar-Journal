### GET 请求的常见用法

```python
# 1. 最简单的GET请求（不带任何参数）
response = http_client.get("/api/doctors")

# 2. 带查询参数的GET请求
response = http_client.get(
    "/api/doctors", 
    params={"department": "内科"}
)

# 3. 带多个查询参数的GET请求
response = http_client.get(
    "/api/doctors",
    params={
        "department": "内科",
        "status": "在职",
        "page": 1,
        "page_size": 10
    }
)
# 实际URL: /api/doctors?department=内科&status=在职&page=1&page_size=10

# 4. 带自定义headers的GET请求
response = http_client.get(
    "/api/doctors",
    headers={"Authorization": "Bearer token123"}
)

# 5. 带查询参数和headers的GET请求
response = http_client.get(
    "/api/doctors",
    params={"department": "内科"},
    headers={"Authorization": "Bearer token123"}
)

# 6. 带自定义超时时间的GET请求
response = http_client.get(
    "/api/doctors",
    timeout=60  # 60秒超时
)

# 7. 完整参数示例（所有参数都用上）
response = http_client.get(
    "/api/doctors",
    params={"department": "内科", "status": "在职"},
    headers={"Authorization": "Bearer token123", "X-Custom-Header": "value"},
    timeout=60
)
```

### POST 请求的常见用法

```python
# 1. 最简单的POST请求（只传必需字段）
medicine_data = {"name": "阿莫西林"}
response = http_client.post("/api/medicines", json_data=medicine_data)

# 2. 带完整数据的POST请求
medicine_data = {
    "name": "阿莫西林胶囊",
    "category": "抗生素",
    "specification": "250mg*24粒",
    "manufacturer": "华北制药",
    "price": 15.50,
    "stock": 100,
    "description": "用于敏感菌所致的各种感染"
}
response = http_client.post("/api/medicines", json_data=medicine_data)

# 3. 带headers的POST请求（比如需要认证）
response = http_client.post(
    "/api/medicines",
    json_data=medicine_data,
    headers={"Authorization": "Bearer token123"}
)

# 4. 带自定义超时的POST请求
response = http_client.post(
    "/api/medicines",
    json_data=medicine_data,
    timeout=60
)

# 5. 完整参数示例
response = http_client.post(
    "/api/medicines",
    json_data=medicine_data,
    headers={"Authorization": "Bearer token123"},
    timeout=60
)
```

### PUT 请求的常见用法

```python
# 1. 更新整个资源（传所有字段）
doctor_data = {
    "doctor_name": "李医生",
    "gender": "女",
    "department": "内科",
    "title": "主治医师",
    "phone": "13900139000",
    "email": "doctor@example.com",
    "status": "在职"
}
response = http_client.put("/api/doctors/1", json_data=doctor_data)

# 2. 更新资源的某个属性（只传要改的字段）
status_data = {"status": "休假"}
response = http_client.put("/api/doctors/1/status", json_data=status_data)

# 3. 带headers的PUT请求
response = http_client.put(
    "/api/doctors/1",
    json_data=doctor_data,
    headers={"Authorization": "Bearer token123"}
)

# 4. 完整参数示例
response = http_client.put(
    "/api/doctors/1",
    json_data=doctor_data,
    headers={"Authorization": "Bearer token123"},
    timeout=60
)
```

response = http_client.delete(

### PATCH 请求的常见用法

```python
# 1. 部分更新（只传要改的字段）
price_update = {"price": 20.0}
response = http_client.patch("/api/medicines/1", json_data=price_update)

# 2. 更新多个字段
update_data = {
    "price": 20.0,
    "stock": 200
}
response = http_client.patch("/api/medicines/1", json_data=update_data)

# 3. 带headers的PATCH请求
response = http_client.patch(
    "/api/medicines/1",
    json_data=update_data,
    headers={"Authorization": "Bearer token123"}
)

# 4. 完整参数示例
response = http_client.patch(
    "/api/medicines/1",
    json_data=update_data,
    headers={"Authorization": "Bearer token123"},
    timeout=60
)
```
